async function createPost() { const postText = document.getElementById('postText').value.trim(); const postButton = document.getElementById('postButton'); const imageFile = document.getElementById('postImage').files[0]; if (!postText && !imageFile) { showMessage('Добавьте текст или фото', 'error'); return; } postButton.disabled = true; postButton.textContent = 'Публикация...'; try { let imageUrl = null; if (imageFile) { const imageRef = storage.ref( `posts/${currentUser.uid}/${Date.now()}_${imageFile.name}` ); await imageRef.put(imageFile); imageUrl = await imageRef.getDownloadURL(); } await db.collection('posts').add({ text: postText, imageUrl: imageUrl, userId: currentUser.uid, username: currentUserData.username, createdAt: new Date(), likes: 0, likedBy: [], commentsCount: 0 }); document.getElementById('postText').value = ''; document.getElementById('postImage').value = ''; showMessage('Пост опубликован!', 'success'); loadPosts(); } catch (error) { console.error(error); showMessage('Ошибка публикации', 'error'); } finally { postButton.disabled = false; postButton.textContent = '📝 Опубликовать'; } }
